    Allemagne: ["Bade-Wurtemberg","Bavière","Berlin","Brandebourg","Brême","Hambourg","Hesse","Mecklembourg-Poméranie-Occidentale","Basse-Saxe","Rhénanie-du-Nord-Westphalie","Rhénanie-Palatinat","Sarre","Saxe","Saxe-Anhalt","Schleswig-Holstein","Thuringe"],
    Italie: ["Piémont","Lombardie","Liguria","Vallée d'Aoste","Toscane","Latium","Vénétie","Émilie-Romagne","Sardaigne","Sicile","Campanie"]
  };

  function escapeHtml(value) {
    return String(value || "")
      .replaceAll("&", "&amp;")
      .replaceAll("<", "&lt;")
      .replaceAll(">", "&gt;")
      .replaceAll('"', "&quot;")
      .replaceAll("'", "&#039;");
  }

  function fillDepartments(country) {
    const departmentSelect = document.getElementById("departmentSelect");
    const regions = regionsByCountry[country] || [];
    departmentSelect.innerHTML = '<option value="">Choisir...</option>';
    regions.forEach(region => {
      const option = document.createElement("option");
      option.value = region;
      option.textContent = region;
      departmentSelect.appendChild(option);
    });
  }

  function renderEvents(events) {
    const results = document.getElementById("eventsResults");

    if (!events || events.length === 0) {
      results.innerHTML = '<div class="notice">Aucun événement trouvé.</div>';
      return;
    }

    results.innerHTML = events.map(event => `
      <div class="event-card">
        ${event.posterurl ? `<img class="poster" src="${escapeHtml(event.posterurl)}" alt="${escapeHtml(event.title || "Affiche événement")}">` : ""}
        <h3>${escapeHtml(event.title || "Événement automobile")}</h3>
        <div class="event-meta"><strong>Date :</strong> ${escapeHtml(event.eventdate || "Non précisée")}</div>
        <div class="event-meta"><strong>Lieu :</strong> ${escapeHtml(event.venuename || "Non précisé")}</div>
        <div class="event-meta">${escapeHtml(event.summary || "")}</div>
        ${event.sourceurl ? `<p><a class="event-link" href="${escapeHtml(event.sourceurl)}" target="_blank" rel="noopener noreferrer">Voir la source</a></p>` : ""}
      </div>
    `).join("");
  }

  document.getElementById("countrySelect").addEventListener("change", function() {
    fillDepartments(this.value);
  });

  document.getElementById("eventSearchForm").addEventListener("submit", async function(e) {
    e.preventDefault();

    const results = document.getElementById("eventsResults");
    const country = document.getElementById("countrySelect").value;
    const department = document.getElementById("departmentSelect").value;

    results.innerHTML = '<div class="notice">Recherche IA en cours...</div>';

    try {
      const edgeBase = window.MONARCH_CONFIG?.EDGE_BASE_URL || "";
      if (!edgeBase) {
        throw new Error("EDGE_BASE_URL non configuré.");
      }

      const functionUrl = `${edgeBase.replace(/\/$/, "")}/ai-events-search`;

      const response = await fetch(functionUrl, {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ country, department })
      });

      const data = await response.json();

      if (!response.ok) {
        throw new Error(data.error || "Erreur lors de la recherche.");
      }

      renderEvents(data.events || []);

    } catch (err) {
      results.innerHTML = `<div class="notice error">${escapeHtml(err.message || "Erreur inconnue.")}</div>`;
      console.error(err);
    }
  });
</script>
